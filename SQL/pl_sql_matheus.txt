-- PL/SQL para inserir
DECLARE
  v_codarea GSAREA.CODAREA%type := 19;
  v_nome GSAREA.NOME%type := 'Teste';
  v_obs varchar2(250) := 'Area responsavel pelo Marketing'; 
BEGIN
  INSERT INTO GSAREA (CODAREA, NOME, OBSERVACAO) VALUES (v_codarea, v_nome, v_obs);
END;

-- PL/SQL para deletar
DECLARE
  v_codarea number(5) := 16;
BEGIN
  DELETE FROM GSAREA WHERE CODAREA = (v_codarea) ;
END;

-- PL/SQL para atualizar
DECLARE
   v_NOME GSAREA.NOME%type := 'Exeperiência do aluno';
   v_CODIGO GSAREA.CODAREA%type := 19;  
BEGIN
  UPDATE GSAREA A SET A.NOME = (v_NOME) WHERE A.CODAREA = (v_CODIGO);
  dbms_output.put_line('Dados atualizados com sucesso!');  
END;
  
-- PL/SQL para selecionar todos os registros
DECLARE
  CURSOR cur_AREA IS
    SELECT A.CODAREA, A.NOME FROM GSAREA A ORDER BY A.CODAREA;
  v_CODAREA GSAREA.CODAREA%type;
  v_NOME    GSAREA.NOME%type;
BEGIN
  OPEN cur_AREA;
  LOOP
    FETCH cur_AREA
      INTO v_CODAREA, v_NOME;
    EXIT WHEN cur_AREA%NOTFOUND;
    dbms_output.put_line('COD: ' || v_CODAREA || ' - NOME DA AREA: ' || v_nome);
  END LOOP;
  CLOSE cur_AREA;
END;


-- Procedure para inclusão
CREATE OR REPLACE PROCEDURE prc_incluir_area(p_CODAREA    IN GSAREA.CODAREA%type,
                                             p_NOME       IN GSAREA.NOME%type,
                                             p_OBSERVACAO IN GSAREA.OBSERVACAO%type) IS
BEGIN
  INSERT INTO GSAREA
    (CODAREA, NOME, OBSERVACAO)
  VALUES
    (p_CODAREA, p_NOME, p_OBSERVACAO);
END prc_incluir_area;

-- Executando a Procedure
begin
  prc_incluir_area(37, 'teste', 'teste');
end;
